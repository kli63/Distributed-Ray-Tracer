90 {
91  Bone *hroot, *bone;
92  bone = hroot = pSkeleton->getRoot();
93 
94  std::ifstream file( name, std::ios::in );
95  if( file.fail() )
96  return -1;
97 
98  int n=0;
99  char str[2048];
100 
101  // count the number of lines
102  while(!file.eof())
103  {
104  file.getline(str, 2048);
105  if(file.eof()) break;
106  //We do not want to count empty lines
107  if (strcmp(str, "") != 0)
108  n++;
109  }
110 
111  file.close();
112 
113  //Compute number of frames.
114  //Subtract 3 to ignore the header
115  //There are (NUM_BONES_IN_ASF_FILE - 2) moving bones and 2 dummy bones (lhipjoint and rhipjoint)
116  int numbones = pSkeleton->numBonesInSkel(bone[0]);
117  int movbones = pSkeleton->movBonesInSkel(bone[0]);
118  n = (n-3)/((movbones) + 1);
119 
120  m_NumFrames = n;
121 
122  //Allocate memory for state bonevector
123  m_pPostures = new Posture[m_NumFrames];
124 
125  //Set all postures to default posture
126  SetPosturesToDefault();
127 
128  file.open(name);
129 
130  // process the header (add rotational DOFs to skeleton if requested)
131  while (1)
132  {
133  file >> str;
134 
135  if(strcmp(str, ":FORCE-ALL-JOINTS-BE-3DOF") == 0)
136  pSkeleton->enableAllRotationalDOFs();
137 
138  if(strcmp(str, ":DEGREES") == 0)
139  break;
140  }
141 
142  for(int i=0; i<m_NumFrames; i++)
143  {
144  //read frame number
145  int frame_num;
146  file >> frame_num;
147 
148  //There are (NUM_BONES_IN_ASF_FILE - 2) movable bones and 2 dummy bones (lhipjoint and rhipjoint)
149  for(int j=0; j<movbones; j++)
150  {
151  //read bone name
152  file >> str;
153 
154  //fine the bone index corresponding to the bone name
155  int bone_idx;
156  for( bone_idx = 0; bone_idx < numbones; bone_idx++ )
157  if( strcmp( str, pSkeleton->idx2name(bone_idx) ) == 0 )
158  break;
159 
160  //init rotation angles for this bone to (0, 0, 0)
161  m_pPostures[i].bone_rotation[bone_idx].setValue(0.0, 0.0, 0.0);
162 
163  for(int x = 0; x < bone[bone_idx].dof; x++)
164  {
165  double tmp;
166  file >> tmp;
167  // printf("%d %f\n",bone[bone_idx].dofo[x],tmp);
168  switch (bone[bone_idx].dofo[x])
169  {
170  case 0:
171  printf("FATAL ERROR in bone %d not found %d\n",bone_idx,x);
172  x = bone[bone_idx].dof;
173  break;
174  case 1:
175  m_pPostures[i].bone_rotation[bone_idx].p[0] = tmp;
176  break;
177  case 2:
178  m_pPostures[i].bone_rotation[bone_idx].p[1] = tmp;
179  break;
180  case 3:
181  m_pPostures[i].bone_rotation[bone_idx].p[2] = tmp;
182  break;
183  case 4:
184  m_pPostures[i].bone_translation[bone_idx].p[0] = tmp * scale;
185  break;
186  case 5:
187  m_pPostures[i].bone_translation[bone_idx].p[1] = tmp * scale;
188  break;
189  case 6:
190  m_pPostures[i].bone_translation[bone_idx].p[2] = tmp * scale;
191  break;
192  case 7:
193  m_pPostures[i].bone_length[bone_idx].p[0] = tmp;// * scale;
194  break;
195  }
196  }
197  if( strcmp( str, "root" ) == 0 )
198  {
199  m_pPostures[i].root_pos.p[0] = m_pPostures[i].bone_translation[0].p[0];// * scale;
200  m_pPostures[i].root_pos.p[1] = m_pPostures[i].bone_translation[0].p[1];// * scale;
201  m_pPostures[i].root_pos.p[2] = m_pPostures[i].bone_translation[0].p[2];// * scale;
202  }
203 
204  // read joint angles, including root orientation
205  }
206  }
207 
208  file.close();
209  printf("%d samples in '%s' are read.\n", n, name);
210  return n;
211 }
