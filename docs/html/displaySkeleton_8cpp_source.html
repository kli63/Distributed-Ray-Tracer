103 /*
104  Define M_k = Modelview matrix at the kth node (bone) in the heirarchy
105  M_k stores the transformation matrix of the kth bone in world coordinates
106  Our goal is to draw the (k+1)th bone, using its local information and M_k
107 
108  In the k+1th node, compute the following matrices:
109  rot_parent_current: this is the rotation matrix that
110  takes us from k+1 to the kth local coordinate system
111  R_k+1 : Rotation matrix for the k+1 th node (bone)
112  using angles specified by the AMC file in local coordinates
113  T_k+1 : Translation matrix for the k+1th node
114 
115  The update relation is given by:
116  M_k+1 = M_k * (rot_parent_current) * R_k+1 + T_k+1
117 */

119 void DisplaySkeleton::DrawBone(Bone *pBone,int skelNum)
120 {
121  static double z_dir[3] = {0.0, 0.0, 1.0};
122  double r_axis[3], theta;
123 
124  //currentMatrix = getCurrentModelview();
125 
126  //Transform (rotate) from the local coordinate system of this bone to it's parent
127  //This step corresponds to doing: ModelviewMatrix = M_k * (rot_parent_current)
128  myMultMatrixd((double*)&pBone->rot_parent_current);
129 
130  //translate AMC (rarely used)
131  if(pBone->doftz)
132  myTranslatef(0.0f, 0.0f, float(pBone->tz));
133  if(pBone->dofty)
134  myTranslatef(0.0f, float(pBone->ty), 0.0f);
135  if(pBone->doftx)
136  myTranslatef(float(pBone->tx), 0.0f, 0.0f);
137 
138  //rotate AMC
139  if(pBone->dofrz)
140  myRotatef(float(pBone->rz), 0.0f, 0.0f, 1.0f);
141  if(pBone->dofry)
142  myRotatef(float(pBone->ry), 0.0f, 1.0f, 0.0f);
143  if(pBone->dofrx)
144  myRotatef(float(pBone->rx), 1.0f, 0.0f, 0.0f);
145 
146  //Store the current ModelviewMatrix (before adding the translation part)
147  myPushMatrix();
148 
149  //Compute tx, ty, tz : translation from pBone to its child (in local coordinate system of pBone)
150  double tx = pBone->dir[0] * pBone->length;
151  double ty = pBone->dir[1] * pBone->length;
152  double tz = pBone->dir[2] * pBone->length;
153 
154  // Use the current ModelviewMatrix to display the current bone
155  // Rotate the bone from its canonical position (elongated sphere
156  // with its major axis parallel to X axis) to its correct orientation
157  if(pBone->idx == Skeleton::getRootIndex())
158  {
159  // do nothing
160  }
161  else
162  {
163  //MATRIX4 currentTransform = getCurrentModelview();
164  MATRIX4 currentTransform = currentMatrix;
165 
166  //Compute the angle between the canonical pose and the correct orientation
167  //(specified in pBone->dir) using cross product.
168  //Using the formula: r_axis = z_dir x pBone->dir
169  r_axis[0] = z_dir[1]*pBone->dir[2]-z_dir[2]*pBone->dir[1];
170  r_axis[1] = z_dir[2]*pBone->dir[0]-z_dir[0]*pBone->dir[2];
171  r_axis[2] = z_dir[0]*pBone->dir[1]-z_dir[1]*pBone->dir[0];
172 
173  double dot_prod = z_dir[0] * pBone->dir[0] + z_dir[1] * pBone->dir[1] + z_dir[2] * pBone->dir[2] ;
174  double r_axis_len = sqrt(r_axis[0] * r_axis[0] + r_axis[1] * r_axis[1] + r_axis[2] * r_axis[2]);
175  theta = atan2(r_axis_len, dot_prod);
176  VEC3 axis(r_axis[0], r_axis[1], r_axis[2]);
177  axis.normalize();
178  MATRIX3 rotation;
179  rotation = AngleAxisd(theta, axis);
180  rotation.transposeInPlace();
181 
182  MATRIX4 scaling = MATRIX4::Identity();
183  scaling(0,0) = pBone->aspx;
184  scaling(1,1) = pBone->aspy;
185  boneScalings[pBone->idx] = scaling;
186 
187  currentTransform = scaling * toMatrix4(rotation) * currentTransform;
188 
189  VEC4 currentTranslation;
190  currentTranslation[0] = currentTransform(3,0);
191  currentTranslation[1] = currentTransform(3,1);
192  currentTranslation[2] = currentTransform(3,2);
193  currentTransform(3,0) = 0;
194  currentTransform(3,1) = 0;
195  currentTransform(3,2) = 0;
196 
197  boneRotations[pBone->idx] = currentTransform.transpose();
198  boneTranslations[pBone->idx] = currentTranslation;
199  }
200 
201  myPopMatrix();
202 
203  // Finally, translate the bone, depending on its length and direction
204  // This step corresponds to doing: M_k+1 = ModelviewMatrix += T_k+1
205  myTranslatef(float(tx), float(ty), float(tz));
206 }
