435 void renderFrames(int start, int end, float* pixels)
436 {
437  for (int x = 0; x< 2400; x += 8) {
438 
439  setSkeletonsToSpecifiedFrame(x);
440  buildScene();
441 
442  double radius = distance(cam.eye, cam.lookat);
443  double angle = (double(x) / 2400.0) * 2.0 * M_PI;
444  cam.lookat = sphereCenters[0];
445  cam.eye[0] = cam.lookat[0] - radius * cos(angle);
446  cam.eye[2] = cam.lookat[2] + radius * sin(angle);
447  cam.update();
448  if (x / 8 < start || x / 8 >= end) {
449  continue;
450  }
451  // cout << cam.focalLength << endl;
452  if (x!= 0 && x < 840)
453  {
454  cam.focalLength = min(cam.focalLength + ((x / 8) * 0.0443), 5.0);
455  // cout << cam.focalLength << endl;
456  cam.aperture = max(cam.aperture - (x / 8) * 0.0005, 0.01);
457  }
458  if ( x > 840 && x < 1680)
459  {
460  cam.focalLength = 5.0;
461  cam.aperture = 0.01;
462  }
463  if (x > 1680)
464  {
465  cam.focalLength = 5.0;
466  cam.aperture = 0.0;
467  }
468  // cout << cam.focalLength << endl;
469  // cout << cam.aperture << endl;
470  bvh_node world_bvh(world);
471 
472  char buffer[256];
473  snprintf(buffer, 256, "./frames/frame.%04i.ppm", x / 8);
474  // renderImage(windowWidth, windowHeight, buffer);
475  auto start = chrono::high_resolution_clock::now();
476  cam.render(world_bvh, lights, pixels, antialiasing, softshadow, depth_of_view);
477  writePPM(buffer, cam.xRes, cam.yRes, pixels);
478  auto stop = chrono::high_resolution_clock::now();
479  auto duration = std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
480  cout << "Rendered " + to_string(x / 8) + " frames using " << duration.count() / 1000000 << " seconds" << endl;
481  // cout << cam.lookat << endl << endl;
482  cam.clearInputs(pixels);
483 
484  // break;
485  }
