19 static MATRIX4 RotationX(float theta)
20 {
21  MATRIX4 R;
22  R.setZero();
23  R(0,0) = R(3,3) = 1;
24  float c = cos(theta);
25  float s = sin(theta);
26  R(1,1) = R(2,2) = c;
27  R(1,2) = -s;
28  R(2,1) = s;
29  return R;
30 }

370 /*
371 This function sets rot_parent_current data member.
372 Rotation from this bone local coordinate system
373 to the coordinate system of its parent
374 */
375 void Skeleton::compute_rotation_parent_child(Bone *parent, Bone *child)
376 {
377  if(child != NULL)
378  {
379  MATRIX4 Rz = RotationZ(-parent->axis_z * M_PI / 180.0);
380  MATRIX4 Ry = RotationY(-parent->axis_y * M_PI / 180.0);
381  MATRIX4 Rx = RotationX(-parent->axis_x * M_PI / 180.0);
382 
383  MATRIX4 tmp1 = Rx * Ry * Rz;
384 
385  Rz = RotationZ(child->axis_z * M_PI / 180.0);
386  Ry = RotationY(child->axis_y * M_PI / 180.0);
387  Rx = RotationX(child->axis_x * M_PI / 180.0);
388 
389  MATRIX4 tmp2 = Rz * Ry * Rx;
390 
391  MATRIX4 tmp = tmp1 * tmp2;
392 
393  for (int x = 0; x < 4; x++)
394  for (int y = 0; y < 4; y++)
395  child->rot_parent_current[x][y] = tmp(y,x);
396  }
397 }

401 // loop through all bones to calculate local coordinate's direction vector and relative orientation
402 void Skeleton::ComputeRotationToParentCoordSystem(Bone *bone)
403 {
404  //Compute tmp2, a matrix containing root
405  //joint local coordinate system orientation
406  int root = Skeleton::getRootIndex();
407  MATRIX4 Rz = RotationZ(bone[root].axis_z * M_PI / 180.0);
408  MATRIX4 Ry = RotationY(bone[root].axis_y * M_PI / 180.0);
409  MATRIX4 Rx = RotationX(bone[root].axis_x * M_PI / 180.0);
410 
411  MATRIX4 tmp2 = Rz * Ry * Rx;
412 
413  for (int x = 0; x < 4; x++)
414  for (int y = 0; y < 4; y++)
415  bone[root].rot_parent_current[x][y] = tmp2(y,x);
416 
417  //Compute rot_parent_current for all other bones
418  int numbones = numBonesInSkel(bone[0]);
419  for(int i=0; i<numbones; i++)
420  {
421  if(bone[i].child != NULL)
422  {
423  compute_rotation_parent_child(&bone[i], bone[i].child);
424 
425  // compute parent child siblings...
426  Bone * tmp = NULL;
427  if (bone[i].child != NULL) tmp = (bone[i].child)->sibling;
428  while (tmp != NULL)
429  {
430  compute_rotation_parent_child(&bone[i], tmp);
431  tmp = tmp->sibling;
432  }
433  }
434  }
435 }

437 /*
438 Transform the direction vector (dir),
439 which is defined in character's global coordinate system in the ASF file,
440 to local coordinate
441 */
442 void Skeleton::RotateBoneDirToLocalCoordSystem()
443 {
444  for(int i=1; i<NUM_BONES_IN_ASF_FILE; i++)
445  {
446  VEC4 dir;
447  dir[0] = m_pBoneList[i].dir[0];
448  dir[1] = m_pBoneList[i].dir[1];
449  dir[2] = m_pBoneList[i].dir[2];
450  dir[3] = 1;
451 
452  MATRIX4 Rz = RotationZ(-m_pBoneList[i].axis_z * M_PI / 180.0);
453  MATRIX4 Ry = RotationY(-m_pBoneList[i].axis_y * M_PI / 180.0);
454  MATRIX4 Rx = RotationX(-m_pBoneList[i].axis_x * M_PI / 180.0);
455 
456  dir = Rx * Ry * Rz * dir;
457  m_pBoneList[i].dir[0] = dir[0];
458  m_pBoneList[i].dir[1] = dir[1];
459  m_pBoneList[i].dir[2] = dir[2];
460  }
